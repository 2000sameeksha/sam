#include<stdio.h>
#include<math.h>
int main()
{
    int i,n,a[10],b[10];
    printf("Enter the number of elements\n");
    scanf("%d",&n);
    printf("Enter the array elements\n");
    for(i=0;i<n;i++)
    {
        scanf("%d",&a[i]);
    }
    for(i=0;i<n;i++)
    {
        b[i]=a[n-1-i];
    }
printf("After swapping\n");
for(i=0;i<n;i++)
{
    printf("%d\t",b[i]);
}
    return 0;
}
