#include<stdio.h>
int main()
{
    char str1[100],str2[100];
    int i=0,j=0;
    printf("Enter a  first string\n");
    scanf("%s",str1);
    printf("Enter a second string\n");
    scanf("%s",str2);

    while(str1[i]!='\0')
    {
        i++;
    }

   while(str2[j]!='\0')
   {
       str1[i]=str2[j];
       j++;
       i++;
   }
   str1[i]='\0';
   printf("Strings after concatenation %s\n",str1);
    return 0;
}
